from datetime import datetime
def format_ts(ts):
    ts_str = ts.isoformat()
    if ts_str.endswith('+00:00'):
        ts_str = ts_str[:-6] + 'Z'
    if '.' in ts_str:
        base, frac = ts_str[:-1].split('.')
        frac = frac.rstrip('0')
        if frac:
            ts_str = f'{base}.{frac}Z'
        else:
            ts_str = f'{base}Z'
    return ts_str

print(format_ts(datetime.fromisoformat('2023-01-01T12:00:00+00:00')))
print(format_ts(datetime.fromisoformat('2023-01-01T12:00:10.001+00:00')))
print(format_ts(datetime.fromisoformat('2023-01-01T12:00:10.123456+00:00')))
