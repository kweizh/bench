In this task you will use the official https://developers.apideck.com/sdks/python.md to seed several tickets in the preconfigured GitHub-backed collection and then walk the entire collection with a small page size to collect every ticket created by the current run. import os
from apideck_unify import Apideck

api_key = os.environ.get('APIDECK_API_KEY')
app_id = os.environ.get('APIDECK_APP_ID')
consumer_id = os.environ.get('APIDECK_CONSUMER_ID')
collection_id = os.environ.get('APIDECK_ISSUE_TRACKING_COLLECTION_ID')
run_id = os.environ.get('ZEALT_RUN_ID')

sdk = Apideck(api_key=api_key, app_id=app_id, consumer_id=consumer_id)

res = sdk.issue_tracking.collection_tickets.list(
    collection_id=collection_id,
    service_id='github',
    limit=2
)

page_count = 0
current_page = res
all_tickets = []
while current_page is not None:
    page_count += 1
    tickets = current_page.get_tickets_response.data or []
    for t in tickets:
        if run_id in (t.subject or ''):
            all_tickets.append(t)
    current_page = current_page.next()

print('Page count:', page_count)
print('Found run-scoped tickets:', len(all_tickets))
for t in all_tickets:
    print(t.id, t.subject)
