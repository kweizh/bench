The unified `List Tickets` endpoint (`GET /issue-tracking/collections/{collection_id}/tickets`) is **cursor-paginated**: each response contains `meta.cursors.next` which must be followed (rather than computing offsets) until exhausted. In this task you will use the official https://developers.apideck.com/sdks/python.md to seed several tickets in the preconfigured GitHub-backed collection and then walk the entire collection with a small page size to collect every ticket created by the current run. import json, os
from pathlib import Path
from apideck_unify import Apideck
p=Path('/home/user/myproject/tickets.json')
data=json.loads(p.read_text())
assert data['run_id'] == os.environ['ZEALT_RUN_ID']
assert data['collection_id'] == os.environ['APIDECK_ISSUE_TRACKING_COLLECTION_ID']
assert isinstance(data['page_count'], int) and data['page_count'] > 0
assert [t['index'] for t in data['tickets']] == [1,2,3,4,5]
client=Apideck(api_key=os.environ['APIDECK_API_KEY'], app_id=os.environ['APIDECK_APP_ID'], consumer_id=os.environ['APIDECK_CONSUMER_ID'])
for t in data['tickets']:
    expected=f"Pagination demo {t['index']} - {data['run_id']}"
    assert t['subject'] == expected
    response=client.issue_tracking.collection_tickets.get(service_id='github', collection_id=data['collection_id'], ticket_id=t['id'])
    payload=response.collection_ticket_response if hasattr(response, 'collection_ticket_response') else getattr(response, 'get_ticket_response', None)
    if payload is None:
        # inspect possible response payload field names without printing secrets
        candidates=[name for name in response.model_fields if name.endswith('_response')]
        raise AssertionError(f'No ticket payload for {t["id"]}; candidates={candidates}')
    server_ticket=payload.data
    assert server_ticket.subject == expected, (t['id'], server_ticket.subject, expected)
print('tickets.json parsed and all 5 tickets verified by Get Ticket')
