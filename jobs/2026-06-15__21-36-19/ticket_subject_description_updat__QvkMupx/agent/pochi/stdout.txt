import json, os, urllib.parse, urllib.request
base='https://unify.apideck.com'
collection=os.environ['APIDECK_ISSUE_TRACKING_COLLECTION_ID']
run=os.environ['ZEALT_RUN_ID']
headers={'Authorization':'Bearer '+os.environ['APIDECK_API_KEY'],'x-apideck-app-id':os.environ['APIDECK_APP_ID'],'x-apideck-consumer-id':os.environ['APIDECK_CONSUMER_ID'],'x-apideck-service-id':'github','Accept':'application/json'}
def get(path):
    req=urllib.request.Request(base+path, headers=headers, method='GET')
    with urllib.request.urlopen(req, timeout=30) as r:
        return json.loads(r.read().decode())
tickets=[]; cursor=None; coll=urllib.parse.quote(collection, safe='')
while True:
    path=f'/issue-tracking/collections/{coll}/tickets'
    if cursor: path += '?cursor=' + urllib.parse.quote(cursor, safe='')
    resp=get(path)
    tickets += [t for t in resp.get('data', []) if isinstance(t, dict)]
    cursor=((resp.get('meta') or {}).get('cursors') or {}).get('next')
    if not cursor: break
v1=[t for t in tickets if run in str(t.get('subject','')) and '[UPDATE-V1]' in str(t.get('subject',''))]
