filters = [
    {'body_metadata': {'group_name': 'support'}},
    {'body_metadata': {'group_name': ['support']}},
    {'body_metadata': {'groupName': 'support'}},
    {'body_metadata': {'groupName': ['support']}},
    {'extra_body': {'metadata': {'group_name': 'support'}}},
    {'extra_body': {'metadata': {'groupName': 'support'}}},
    {'extra_body': {'metadata': {'group_name': ['support']}}},
    {'extra_body': {'metadata': {'groupName': ['support']}}}
]

for f in filters:
    try:
        res = client.v1.context.search(
            query='support',
            similarity_threshold=0.5,
            minimum_similarity_threshold=0.1,
            metadata='true',
            **f
        )
        groups_found = set()
        for ctx in res.contexts:
            g = ctx.metadata.get('group_name') if ctx.metadata else None
            if g:
                groups_found.update(g)
        print(f'Filter {f}: success, groups found: {groups_found}')
    except Exception as e:
        print(f'Filter {f}: error: {e}')
