from bytewax.inputs import FixedPartitionedSource, StatelessSourcePartition

class FileSource(FixedPartitionedSource):
    def __init__(self, path):
        self._path = path

    def list_parts(self):
        return [str(self._path)]

    def build_part(self, step_id, path, resume_state):
        class _Part(StatelessSourcePartition):
            def __init__(self, path):
                self._path = path
                self._f = None

            def next_batch(self):
                if self._f is None:
                    self._f = open(self._path)
                line = self._f.readline()
                if not line:
                    raise StopIteration()
                return [line.strip()]

            def close(self):
                if self._f:
                    self._f.close()

        return _Part(path)
