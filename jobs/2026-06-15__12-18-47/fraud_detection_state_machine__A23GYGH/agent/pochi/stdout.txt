import argparse
import json
from bytewax.dataflow import Dataflow
import bytewax.operators as op
from bytewax.connectors.files import FileSource, FileSink
from bytewax.testing import run_main

def parse_args():
    parser = argparse.ArgumentParser()
    parser.add_argument("--input", required=True)
    parser.add_argument("--output", required=True)
    return parser.parse_args()

def state_machine(state, event):
    if state is None:
        state = {"state": "LOGGED_OUT", "login_ts": None, "large_tx_count": 0}
    
    new_state = state.copy()
    event_type = event["event_type"]
    ts = event["timestamp"]
    
    if event_type == "login":
        new_state["state"] = "LOGGED_IN"
        new_state["login_ts"] = ts
        new_state["large_tx_count"] = 0
        return new_state, []
        
    elif event_type == "logout":
        new_state["state"] = "LOGGED_OUT"
        new_state["login_ts"] = None
        new_state["large_tx_count"] = 0
        return new_state, []
        
    elif event_type == "transaction":
        if new_state["state"] in ("LOGGED_IN", "SUSPICIOUS"):
            if ts - new_state["login_ts"] > 300:
                new_state["state"] = "LOGGED_OUT"
                new_state["login_ts"] = None
                new_state["large_tx_count"] = 0
                return new_state, []
            
            amount = event.get("amount", 0)
            if amount >= 1000:
                new_state["large_tx_count"] += 1
                new_state["state"] = "SUSPICIOUS"
                
                if new_state["large_tx_count"] >= 3:
                    new_state["state"] = "LOGGED_OUT"
                    new_state["login_ts"] = None
                    new_state["large_tx_count"] = 0
                    return new_state, ["FRAUD_ALERT"]
                    
        return new_state, []
    
    return new_state, []

def format_output(key_alert):
    key, alert = key_alert
    return json.dumps({"user_id": key, "alert": alert})

def main():
    args = parse_args()
    
    flow = Dataflow("fraud_detection")
    
    # Read from input file
    inp = op.input("inp", flow, FileSource(args.input))
    
    # Parse JSON
    parsed = op.map("parse_json", inp, json.loads)
    
    # Key by user_id
    keyed = op.key_on("key_on_user", parsed, lambda x: x["user_id"])
    
    # Apply state machine
    processed = op.stateful_map("fraud_state_machine", keyed, state_machine)
    
    # Flatten the alerts (removes empty lists, flattens lists of alerts)
    alerts = op.flat_map_value("flatten_alerts", processed, lambda x: x)
    
    # Format to JSON strings
    formatted = op.map("format_output", alerts, format_output)
    
    # Write to output file
    op.output("out", formatted, FileSink(args.output))
    
    run_main(flow)

if __name__ == "__main__":
    main()

### Implementation Details:
- **State Machine**: The state is maintained uniquely for each `user_id` using Bytewax's `op.stateful_map`. The state is represented as a dictionary and copied (to ensure immutability as per Bytewax guidelines) on every transition.
- **Event Handling**: 
  - On `login`, the state transitions to `LOGGED_IN` and records the timestamp.
  - On `transaction`, if the user is in `LOGGED_IN` or `SUSPICIOUS`, it checks if the time elapsed since login is `> 300` seconds. If so, it resets to `LOGGED_OUT`. Otherwise, if the amount is `>= 1000`, it increments the large transaction count and transitions to `SUSPICIOUS`.
  - Once the count reaches 3, it emits a `FRAUD_ALERT` and resets the state to `LOGGED_OUT`.
  - On `logout`, the user state immediately resets to `LOGGED_OUT`.
- **Dataflow Setup**:
  - `FileSource` reads the input JSONlines.
  - `op.map` parses the JSON strings into dictionaries.
  - `op.key_on` turns the stream into a keyed stream using `user_id` as the key.
  - `op.stateful_map` applies the state machine logic, yielding a tuple of `(new_state, [alerts])`.
  - `op.flat_map_value` flattens the alerts list (which filters out empty lists when no alert is generated).
  - `op.map` formats the output into a JSON string while keeping the tuple structure `(key, json_string)` required by `FileSink`.
  - `FileSink` writes the resulting JSONlines to the output file.
