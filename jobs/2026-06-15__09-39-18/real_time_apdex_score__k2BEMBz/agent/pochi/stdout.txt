from datetime import datetime, timedelta, timezone
from bytewax.dataflow import Dataflow
import bytewax.operators as op
from bytewax.operators.windowing import EventClock, TumblingWindower, fold_window
from bytewax.testing import TestingSource

flow = Dataflow('test')

data = [
    ('auth', {'response_time_ms': 200, 'timestamp': '2023-10-01T10:00:00Z'}, datetime(2023, 10, 1, 10, 0, 0, tzinfo=timezone.utc)),
    ('auth', {'response_time_ms': 600, 'timestamp': '2023-10-01T10:00:05Z'}, datetime(2023, 10, 1, 10, 0, 5, tzinfo=timezone.utc)),
]

inp = op.input('inp', flow, TestingSource(data))
keyed = op.key_on('key', inp, lambda x: x[0])

clock = EventClock(
    ts_getter=lambda x: x[2],
    wait_for_system_duration=timedelta(seconds=0),
)

windower = TumblingWindower(
    length=timedelta(seconds=10),
    align_to=datetime(2023, 1, 1, 0, 0, 0, tzinfo=timezone.utc),
)

windowed = fold_window(
    'fold',
    keyed,
    clock,
    windower,
    builder=lambda: 0,
    folder=lambda acc, x: acc + x[1]['response_time_ms'],
    merger=lambda a, b: a + b,
)

op.inspect('down', windowed.down)
op.inspect('meta', windowed.meta)

from bytewax.testing import run_main
run_main(flow)
