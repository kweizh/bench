import os
from daytona import Daytona, CreateSandboxFromSnapshotParams

def main():
    run_id = "test-123"
    sandbox_name = f"fs-py-{run_id}"
    
    daytona = Daytona()
    sandbox = daytona.create(CreateSandboxFromSnapshotParams(name=sandbox_name, language="python"))
    print("Sandbox created:", sandbox.name)
    try:
        content = f"Hello Daytona {run_id}".encode("utf-8")
        remote_path = "/workspace/input.txt"
        
        # Upload
        sandbox.fs.upload_file(content, remote_path)
        print("Uploaded")
        
        # Exec
        remote_out_path = "/workspace/output.txt"
        resp = sandbox.process.exec(f"tr '[:lower:]' '[:upper:]' < {remote_path} > {remote_out_path}")
        print("Exec response:", resp.exit_code, resp.result)
        
        # Download
        out_bytes = sandbox.fs.download_file(remote_out_path)
        print("Downloaded:", out_bytes)
        
    finally:
        daytona.delete(sandbox)
        print("Sandbox deleted")

if __name__ == "__main__":
    main()
