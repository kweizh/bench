import threading
import requests
import time
import sys

TOKEN = "REDACTED"
URL = "http://localhost:8090/api/wallets/transfer"
FROM_ID = "x8fqd2xayvmiupa"
TO_ID = "8nzg5qtvsacgn3w"
AMOUNT = 1
NUM_TRANSFERS = 50
NUM_WORKERS = 10

results = []
lock = threading.Lock()

def do_transfer(task_id):
    try:
        resp = requests.post(
            URL,
            json={"fromId": FROM_ID, "toId": TO_ID, "amount": AMOUNT},
            headers={"Authorization": TOKEN, "Content-Type": "application/json"},
            timeout=30
        )
        with lock:
            results.append((task_id, resp.status_code, resp.json()))
    except Exception as e:
        with lock:
            results.append((task_id, -1, str(e)))

start = time.time()

# Use a semaphore to limit to NUM_WORKERS concurrent requests
semaphore = threading.Semaphore(NUM_WORKERS)
threads = []

def worker(task_id):
    with semaphore:
        do_transfer(task_id)

for i in range(NUM_TRANSFERS):
    t = threading.Thread(target=worker, args=(i,))
    threads.append(t)
    t.start()

for t in threads:
    t.join()

elapsed = time.time() - start

# Analyze results
status_counts = {}
for (tid, status, body) in results:
    status_counts[status] = status_counts.get(status, 0) + 1

print(f"Completed {len(results)} requests in {elapsed:.2f}s")
print(f"Status code distribution: {status_counts}")

non_200 = [(tid, s, b) for (tid, s, b) in results if s != 200]
if non_200:
    print(f"Non-200 responses ({len(non_200)}):")
    for (tid, s, b) in non_200[:5]:
        print(f"  task={tid} status={s} body={b}")

ok = (
    status_counts.get(200, 0) == NUM_TRANSFERS and
    elapsed < 30
)
print(f"\nAll {NUM_TRANSFERS} returned 200: {status_counts.get(200,0) == NUM_TRANSFERS}")
print(f"Elapsed under 30s: {elapsed < 30}")
sys.exit(0 if ok else 1)

## What was built

**`/home/user/myproject/main.go`** — complete replacement with the transfer handler registered on `OnServe()`.

### Key design decisions

| Concern | Solution |
|---|---|
| **Authentication** | `e.Auth == nil` → HTTP 401 |
| **Deadlock prevention** | Both wallet IDs are fetched inside the transaction in **id-ascending order** — guarantees a global lock ordering regardless of `fromId`/`toId` direction |
| **SQLITE_BUSY retry** | `withRetry()` wraps every transaction: up to 5 attempts, truncated exponential back-off (50 ms → 1.6 s) plus random ≤25 ms jitter so concurrent goroutines don't pile up in lock-step |
| **Atomicity** | `app.RunInTransaction` — both balance updates **and** the audit insert succeed or roll back together |
| **Insufficient funds** | Transaction is rolled back; HTTP 400 + `{"error":"insufficient funds"}` is returned; wallets are unchanged |
