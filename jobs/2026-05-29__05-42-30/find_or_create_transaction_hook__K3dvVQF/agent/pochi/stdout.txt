User.addHook('afterCreate', async (user) => {

    throw new Error('Simulated failure');
  }

async function runFindOrCreate(username) {
  try {
    const result = await sequelize.transaction(async (t) => {
      const [user, created] = await User.findOrCreate({
        where: { username },
        defaults: { status: 'active' },
        transaction: t,
      });
      return user;
    });
    return result;
  } catch (err) {
    return err;
  }
}

**Key design decision — `runFindOrCreate`:**
Sequelize's `findOrCreate` creates an internal transaction, but its v6 implementation has a bug where the `finally` block always calls `transaction.commit()` even when an error occurs, preventing proper rollback. To ensure atomic rollback, `runFindOrCreate` wraps the call in a managed transaction (`sequelize.transaction(async (t) => {...})`) and passes that transaction to `findOrCreate`. When the `afterCreate` hook throws, the managed transaction rolls back both the `User` and `AuditLog` entries.
