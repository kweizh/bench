const limit = 3;
const windowSecs = 60;

routerUse((e) => {
    if (e.request.method === "POST" && e.request.url.path === "/api/collections/users/records") {
        try {
            const ip = e.realIP();
            const now = Math.floor(Date.now() / 1000);
            
            let rawHistory = $app.store().get(ip);
            let history = [];
            if (rawHistory && rawHistory.length) {
                for (let i = 0; i < rawHistory.length; i++) {
                    history.push(rawHistory[i]);
                }
            }
            
            history = history.filter(t => now - t < windowSecs);
            
            if (history.length >= limit) {
                const oldest = history[0];
                let retryAfter = windowSecs - (now - oldest);
                if (retryAfter < 1) retryAfter = 1;
                
                e.response.header().set("Retry-After", String(retryAfter));
                return e.json(429, { "retryAfter": retryAfter });
            }
            
            history.push(now);
            $app.store().set(ip, history);
            
            return e.next();
        } catch (err) {
            console.log("Rate limiter error:", err);
            return e.next();
        }
    }
    return e.next()
})
