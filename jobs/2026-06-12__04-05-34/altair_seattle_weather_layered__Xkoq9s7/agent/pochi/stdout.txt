import altair as alt
from vega_datasets import data
source_url = data.seattle_weather.url
hover_selection = alt.selection_point(name='hover', on='pointerover', nearest=True, encodings=['x'], empty=False)
base = alt.Chart(source_url).transform_calculate(
    temp_mean='(datum.temp_min + datum.temp_max) / 2',
    tooltip_text=\"'temp'\"
)
area = base.mark_area().encode(x='date:T', y='temp_min:Q', y2='temp_max:Q')
line = base.mark_line().encode(x='date:T', y='temp_mean:Q')
bar = base.mark_bar().encode(x='date:T', y='precipitation:Q')
rule_zero = base.mark_rule().encode(y=alt.datum(0))
hover_rule = base.mark_rule().encode(x='date:T')
hover_text = base.mark_text().encode(x='date:T', y=alt.value(20), text='tooltip_text:N')
chart = alt.layer(area, line, bar, rule_zero, hover_rule, hover_text).resolve_scale(y='independent')
print(chart.to_dict())
