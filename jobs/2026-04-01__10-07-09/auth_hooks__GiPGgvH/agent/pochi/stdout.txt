### `src/auth/hooks.ts`
```ts
import { HttpError } from 'wasp/server'
import type { OnBeforeSignupHook } from 'wasp/server/auth'

export const onBeforeSignup: OnBeforeSignupHook = async ({ providerId }) => {
  if (providerId.providerUserId.includes('blocked')) {
    throw new HttpError(403, 'This username is not allowed.')
  }
}
```
> **Note**: In Wasp 0.21, the hook receives `providerId: { providerName, providerUserId }` â€” not a bare `providerUserId`. The username check uses `providerId.providerUserId`.
