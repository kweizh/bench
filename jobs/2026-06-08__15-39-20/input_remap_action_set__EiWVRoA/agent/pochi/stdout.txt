        "move_left": KEY_A,
        "move_right": KEY_D,
        "interact": KEY_E,

    }
    
    for action in actions:
        var ev = InputEventKey.new()
        ev.keycode = actions[action]
        ProjectSettings.set_setting("input/" + action, {"deadzone": 0.5, "events": [ev]})
    
    ProjectSettings.set_setting("autoload/InputRemapper", "*res://autoloads/InputRemapper.gd")
    ProjectSettings.save()
    quit()
