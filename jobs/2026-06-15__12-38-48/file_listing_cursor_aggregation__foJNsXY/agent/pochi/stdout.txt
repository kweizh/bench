import json, os, requests
headers={
 'Authorization': 'Bearer '+os.environ['APIDECK_API_KEY'],
 'x-apideck-app-id': os.environ['APIDECK_APP_ID'],
 'x-apideck-consumer-id': os.environ['APIDECK_CONSUMER_ID'],
 'x-apideck-service-id': 'onedrive',
}
run_id=os.environ['ZEALT_RUN_ID']; prefix=f'AGG-{run_id}-'
drives=requests.get('https://unify.apideck.com/file-storage/drives',headers=headers,params={'limit':100},timeout=30).json()['data']
drive=next(d for d in drives if d.get('name')==os.environ['APIDECK_FILE_STORAGE_DRIVE_NAME'] or d.get('id')==os.environ['APIDECK_FILE_STORAGE_DRIVE_NAME'])
all_matches=[]; cursor=None; pages=0
while True:
 params={'filter[drive_id]':drive['id'],'filter[folder_id]':'root','limit':3}
 if cursor: params['cursor']=cursor
 payload=requests.get('https://unify.apideck.com/file-storage/files',headers=headers,params=params,timeout=30).json()
 pages += 1
 all_matches += [(x.get('name'),x.get('id')) for x in payload.get('data',[]) if str(x.get('name','')).startswith(prefix)]
 cursor=((payload.get('meta') or {}).get('cursors') or {}).get('next')
 if not cursor: break
log=json.load(open('/home/user/apideck_task/output.log'))
print(json.dumps({'pages_walked_with_limit_3': pages, 'matching_names': sorted(n for n,_ in all_matches), 'count': len(all_matches), 'ids_match_log': set(i for _,i in all_matches)==set(log['ids'])}, indent=2))
