import apideck_unify
import os

# Initialize SDK
sdk = apideck_unify.Apideck(
    api_key=os.environ['APIDECK_API_KEY'],
    consumer_id=os.environ['APIDECK_CONSUMER_ID'],
    app_id=os.environ['APIDECK_APP_ID'],
)

# List drives to find the OneDrive drive
try:
    resp = sdk.file_storage.drives.list(
        service_id='onedrive',
        raw=False,
    )
    print('Response type:', type(resp))
    print('Response:', resp)
    if hasattr(resp, 'get_drives_response'):
        gdr = resp.get_drives_response
        if gdr and hasattr(gdr, 'data'):
            for d in gdr.data:
                print(f'Drive: id={d.id}, name={d.name}')
except Exception as e:
    print(f'Error: {e}')
    print(f'Type: {type(e)}')

import apideck_unify
import os

sdk = apideck_unify.Apideck(
    api_key=os.environ['APIDECK_API_KEY'],
    consumer_id=os.environ['APIDECK_CONSUMER_ID'],
    app_id=os.environ['APIDECK_APP_ID'],
)

# Create upload session for a small text file
try:
    resp = sdk.file_storage.upload_sessions.create(
        name='test.txt',
        parent_folder_id='/',
        size=10,
        service_id='onedrive',
        drive_id='3ada079b78534ff1',
    )
    print('Response:', resp)
    if resp.create_upload_session_response:
        data = resp.create_upload_session_response.data
        print('Session ID:', data.id)
except Exception as e:

    traceback.print_exc()
