package main

import (
	"log"
	"net/http"
	"os"
	"sync"
	"time"

	"github.com/pocketbase/pocketbase"
	"github.com/pocketbase/pocketbase/apis"
	"github.com/pocketbase/pocketbase/core"
)

var (
	mu      sync.Mutex
	clients = make(map[string]*Client)
)

type Client struct {
	Count     int
	ResetTime time.Time
}

func getClient(ip string) *Client {
	mu.Lock()
	defer mu.Unlock()

	client, exists := clients[ip]
	now := time.Now()

	if !exists || now.After(client.ResetTime) {
		client = &Client{
			Count:     0,
			ResetTime: now.Add(1 * time.Minute),
		}
		clients[ip] = client
	}

	return client
}

func main() {
	app := pocketbase.New()

	app.OnServe().BindFunc(func(se *core.ServeEvent) error {
		se.Router.POST("/api/custom_signup", func(e *core.RequestEvent) error {
			ip := e.Request.RemoteAddr
			// e.Request.RemoteAddr has ip:port usually, let's just use it or clean it up.
			// e.RemoteIP() is cleaner.
			remoteIP := e.RemoteIP()

			client := getClient(remoteIP)

			mu.Lock()
			client.Count++
			count := client.Count
			mu.Unlock()

			if count > 5 {
				return e.JSON(http.StatusTooManyRequests, map[string]string{"error": "Too Many Requests"})
			}

			var data struct {
				Email           string `json:"email"`
				Password        string `json:"password"`
				PasswordConfirm string `json:"passwordConfirm"`
			}
			if err := e.BindBody(&data); err != nil {
				return e.BadRequestError("Invalid request", err)
			}

			collection, err := app.FindCollectionByNameOrId("users")
			if err != nil {
				return e.InternalServerError("Failed to find users collection", err)
			}

			record := core.NewRecord(collection)
			record.Set("email", data.Email)
			record.Set("password", data.Password)
			record.Set("passwordConfirm", data.PasswordConfirm)

			if err := app.Save(record); err != nil {
				return e.BadRequestError("Failed to create user", err)
			}

			return e.JSON(http.StatusOK, map[string]interface{}{
				"id": record.Id,
				"email": record.GetString("email"),
			})
		})

		se.Router.GET("/{path...}", apis.Static(os.DirFS("./pb_public"), false))
		return se.Next()
	})

	if err := app.Start(); err != nil {
		log.Fatal(err)
	}
}
