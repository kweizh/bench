import { createWorkspace, teardownWorkspace, Workspace } from '@tigrisdata/agent-kit';
import { S3Client, PutObjectCommand } from '@aws-sdk/client-s3';
import { promises as fs } from 'fs';

const STATE_FILE = '/home/user/pool/pool-state.json';

interface WorkspaceState {
  name: string;
  bucket: string;
  credentials?: {
    accessKeyId: string;
    secretAccessKey: string;
  };
}

interface PoolState {
  workspaces: WorkspaceState[];
}

export async function provisionPool(n: number) {
  const promises = [];
  for (let i = 1; i <= n; i++) {
    const name = `pool-${n}-${i}`;
    promises.push(
      createWorkspace(name, {
        ttl: { days: 1 },
        credentials: { role: 'Editor' },
      }).then((res) => {
        if (res.error) {
          throw res.error;
        }
        return {
          name,
          bucket: res.data.bucket,
          credentials: res.data.credentials,
        };
      })
    );
  }

  const workspaces = await Promise.all(promises);

  const state: PoolState = { workspaces };
  await fs.writeFile(STATE_FILE, JSON.stringify(state, null, 2), 'utf-8');
}

export async function assignTask(index: number, text: string) {
  const data = await fs.readFile(STATE_FILE, 'utf-8');
  const state: PoolState = JSON.parse(data);

  if (index < 1 || index > state.workspaces.length) {
    throw new Error(`Agent index ${index} out of range`);
  }

  const workspace = state.workspaces[index - 1];
  if (!workspace.credentials) {
    throw new Error(`No credentials found for workspace ${workspace.name}`);
  }

  const s3 = new S3Client({
    endpoint: 'REDACTED',
    region: 'REDACTED',
    credentials: {
      accessKeyId: workspace.credentials.accessKeyId,
      secretAccessKey: workspace.credentials.secretAccessKey,
    },
  });

  await s3.send(
    new PutObjectCommand({
      Bucket: workspace.bucket,
      Key: 'task.txt',
      Body: text,
    })
  );
}

export async function status() {
  const data = await fs.readFile(STATE_FILE, 'utf-8');
  const state: PoolState = JSON.parse(data);

  console.log(`pool size: ${state.workspaces.length}`);
  for (const w of state.workspaces) {
    console.log(w.bucket);
  }
}

export async function teardownPool() {
  const data = await fs.readFile(STATE_FILE, 'utf-8');
  const state: PoolState = JSON.parse(data);

  const promises = state.workspaces.map((w) =>
    teardownWorkspace({ bucket: w.bucket })
  );

  const results = await Promise.allSettled(promises);

  const errors = results.filter((r) => r.status === 'rejected' || (r.status === 'fulfilled' && r.value.error));
  if (errors.length > 0) {
    throw new Error('Some teardowns failed');
  }

  await fs.unlink(STATE_FILE);
}
